\section{Overview}


\begin{frame}{Basic Reentry}
    \begin{itemize}
        \item A reentry circuit is a circulating excitation wave in a closed pathway of excitable tissue.
        \item \textbf{Structure:} Loop (anatomical or functional) with cell-to-cell conduction.
        \item \textbf{Sustaining Condition and Key Dynamics:}
        \begin{itemize}
            \item Wave returns after tissue has recovered excitability
            \item Requires sufficient path length relative to conduction delay and refractory period
            \item Favored by tissue heterogeneity (e.g., areas of slow conduction)
        \end{itemize}
    \end{itemize}
    \begin{center}
        \includegraphics[width=0.4\linewidth]{basicReentry.jpg}
        
        \scriptsize{\textit{Figure source: Kaye G C. Percutaneous interventional electrophysiology, BMJ 2003;327:280. doi:10.1136/bmj.327.7409.280}}
    \end{center}
\end{frame}


\begin{frame}{Micro Reentry vs Macro Reentry}
\begin{itemize}
	\item \textbf{Macro-reentry:} \cite{cvphysiology_reentry} 
    \begin{itemize}
		\item Large-scale circuit around anatomical structures (e.g., valves, scars, \textbf{pulmonary veins})
        \item Path length is long relative to wavelength ($\lambda = v \cdot \text{ERP}$)
        \item Typically mappable with clinical techniques (e.g., ECG, catheter mapping)
    \end{itemize}

	\item \textbf{Micro-reentry:} \cite{oliveira2018microreentry}
    \begin{itemize}
        \item Small, localized circuits driven by tissue heterogeneity
        \item Path length comparable to wavelength (near minimum for sustained reentry)
        \item Often below clinical resolution; inferred via modeling or high-density mapping
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Overview of Micro Reentry}
\begin{itemize}
    \item \textbf{Definition:} Micro reentry circuits are small-scale reentrant 
		pathways in excitable media (e.g., cardiac tissue) that sustain 
		self-propagating electrical activity.\cite{aha_microreentry2016}

    \item \textbf{Mechanism:}
    \begin{itemize}
        \item Requires a closed conduction loop
        \item Unidirectional block in part of the pathway
        \item Conduction delay allowing tissue ahead to recover excitability
    \end{itemize}

	\item \textbf{Characteristic Scale:} 
    \begin{itemize}
        \item Typically on the order of centimeters or smaller
        \item Often below spatial resolution of clinical imaging
    \end{itemize}

    \item \textbf{Physiological Context:}
    \begin{itemize}
        \item Commonly observed in cardiac arrhythmias (e.g., atrial 
			fibrillation, ventricular tachycardia)
        \item Associated with heterogeneous tissue properties (fibrosis, ischemia)
    \end{itemize}

\end{itemize}
\vspace{-5pt}

\end{frame}


\begin{frame}{Why Study Reentry?}
\begin{itemize}
    \item \textbf{Primary mechanism of arrhythmias:}
    \begin{itemize}
        \item Reentry underlies many tachyarrhythmias (e.g., atrial fibrillation, 
			ventricular tachycardia) \cite{jalife2000rotors, nattel2002af}
    \end{itemize}

    \item \textbf{Guides therapeutic intervention:}
    \begin{itemize}
        \item Identification of reentrant pathways enables targeted 
			ablation strategies \cite{haissaguerre1998af}
    \end{itemize}

    \item \textbf{Links structure to function:}
    \begin{itemize}
        \item Tissue heterogeneity (fibrosis, infarct) promotes reentry 
			formation and stability \cite{oliveira2018microreentry}
    \end{itemize}

    \item \textbf{Enables predictive modeling:}
    \begin{itemize}
        \item Computational models of reentry support patient-specific 
			diagnosis and treatment planning \cite{trayanova2011modeling}
    \end{itemize}

\end{itemize}
\end{frame}

\begin{frame}{Micro Reentry After Ablation}
    \begin{itemize}
        \item Microreentrant atrial tachycardias can occur after AF ablation, especially at gaps in ablation lines \cite{ng2017microreentrant}.
        \item These circuits are very small—sometimes just a few millimeters—and often missed with traditional mapping \cite{mantovan2020microreentry}.
        \item High-resolution mapping enables detection and targeted ablation of these circuits.
    \end{itemize}
    \begin{center}
        \includegraphics[width=0.8\linewidth]{MicroLeftTach.png}
        
        \scriptsize{\textit{Figure: High-density mapping of a microreentrant circuit near a gap in the ablation line~\cite{ng2017microreentrant}}}
    \end{center}
\end{frame}

\begin{frame}{Size of Microreentrant Circuits}
    \begin{itemize}
        \item Microreentrant circuits can be extremely small—sometimes just a few millimeters, far below conventional mapping resolution and detectable only with high-density systems like Rhythmia-Orion \cite{mantovan2020microreentry,ng2017microreentrant}.
        \item The likelihood and location of microreentry depend on tissue structure, wall thickness, and fibrosis.
    \end{itemize}
    \begin{center}
        \includegraphics[width=0.7\linewidth]{HowSmallMicroReentry.png}
        
        {\scriptsize \textit{Figure: High-density mapping reveals a microreentrant circuit confined to a millimeter-scale region~\cite{mantovan2020microreentry}.}}
    \end{center}
\end{frame}

\section{What we are doing}

\begin{frame}{Our Goal}
\begin{itemize}
    \item \textbf{Objective:}
    \begin{itemize}
        \item Develop a computational framework to automatically detect reentrant activity in excitable networks
    \end{itemize}

    \item \textbf{Input Parameters:}
    \begin{itemize}
        \item Network topology (muscular connectivity graph)
        \item Conduction delays along edges
        \item Node refractory periods
    \end{itemize}

    \item \textbf{Core Goal:}
    \begin{itemize}
        \item Identify conditions under which circulating excitation (reentry) emerges from initial stimulation
    \end{itemize}

    \item \textbf{Technical Focus:}
    \begin{itemize}
        \item Detect closed-loop propagation satisfying timing constraints (return after recovery)
        \item Characterize stability and persistence of reentrant pathways
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{How does this goal fit with the other projects we have?}
    \begin{itemize}
    \item \textbf{Integration with Existing Work:}
    \begin{itemize}
        \item Builds on top of the main simulator to enable a more simplistic way
			to search for reentrany circuit dynamics. The long term goal is 
			to use this tool to determine what the pattern is for reentrant 
			circuits and to make some connections to scar tissue or fibrosis 
			patterns.
    \end{itemize}

    \item \textbf{Shared Parameters, Extended Search Dynamics}
    \begin{itemize}
        \item Utilizes the same network properties such as RP and CT while 
			increasing the generality of the simulation. 
		\item Spatial dimentionality has also been removed to 
			consider solely the connections between nodes and their properties.
		\item Other methods of examing circuits are also enabled such as 
			monte carlo simulations.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Simple mesh program}
\begin{itemize}
    \item \textbf{Simulator Overview:}
    \begin{itemize}
        \item Discrete-time excitable network simulator modeling electrical 
			propagation through cardiac-like tissue
		\item Designed to enable automatic search for network properties-
			currently focused on reentry detection
    \end{itemize}

    \item \textbf{Initialization \& Control:}
    \begin{itemize}
        \item Configurable initial stimulation patterns and network parameters
        \item Supports parameter sweeps across conduction and refractory properties
    \end{itemize}

    \item \textbf{Outputs:}
    \begin{itemize}
        \item Setup to record network properties based on initial configuration
			parameters that result in a reentrant circuit
		\item Any output record can be viewed using just a terminal
    \end{itemize}
\end{itemize}
\end{frame}

\section{Results}
\begin{frame}{Basic Setup}
    \begin{center}
		\includegraphics[width=0.75\linewidth]{BasicBox.png}
    \end{center}
	\vspace{-5pt}
	\begin{itemize}
	\item {We tested several different shapes by hand to see what the smallest 
		shape that would produce a micro reentry circuit would be. 
		We found it to be this box with 2 appendages. We used the basic 
		values of $RP=0.04$ and $CV=0.3$}
	\end{itemize}
\end{frame}
\begin{frame}{Current Results}
	\begin{itemize}
	\item An initial trial run of 1000 Monte Carlo simulations with RP ranges
		of (0.01, 0.1) and a CT range of (3.0,4.0) in our initial simple box 
		yielded a 64.9\% reentry rate.
	\item These results were transferable into the main simulator, although 
		not every successful run was checked due to how long it takes to setup 
		in the main sim
	\item Future Monte Carlo sims will include both exploring different 
		ranges of values and different sets of muscles to modify
	\end{itemize}
\end{frame}


\begin{frame}{Transferring setups from the mini-sim to full model}
    \begin{itemize}
    \item \textbf{Model Integration:}
    \begin{itemize}
        \item Results are stored in JSON format and include the origal 
			paremeters for the monte carlo sim, as well as each trial result
		\item Trial results include the specific parameters for each modified 
			muscle and whether that trial resulted in a reentrant circuit
    \end{itemize}

    \item \textbf{Interface to Simulator:}
    \begin{itemize}
		\item Currently models need to be recreated by hand. Long term it 
			should not be difficult to write a script to generate the files 
			needed to run in the main simulator.
        \item In addition, the mini-sim used CT rather than CV and 
			Length due to not using a spatial model, so this needs to be 
			accounted for when transferring models.
    \end{itemize}

    \item \textbf{Scalability Considerations:}
    \begin{itemize}
        \item Currently designed primarily for relatively small networks. 
			However, due to not including things like contraction, it does not 
			require a significant amount of computational resources
    \end{itemize}
\end{itemize}
\end{frame}
\begin{frame}{Current Limitations}	
	\begin{itemize}
		\item Currently it is difficult to efficiently move results between
			simlulators. It should not be too difficult to fix this in the 
			future.
		\item This simulator is primarily designed for running small scale 
			simulations. It is doubtful that it would perform well for 
			larger meshes. 
	\end{itemize}
\end{frame}
\begin{frame}{Future plans for the project}
	\begin{itemize}
		\item Perform more monte carlo sims across a wider range of 
			sub-networks (That is, differnt sets of muscles that get modified)
		\item Automate the process of moving sims to the main model
		\item Use the simulation results to determine a general rule or pattern 
			for micro circuits in this type of model and use that pattern to 
			draw conculsions about micro reentry circutis in the real heart 
	\end{itemize}
\end{frame}
