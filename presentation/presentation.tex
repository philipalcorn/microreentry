\section{Discrete Cosine Transform}

\begin{frame}{Definition}
The discrete cosine transform (DCT) of a signal is defined as [1], [2]
\vspace{-5pt}

\begin{equation}
\vspace{-5pt}
G_x(k)=\dfrac{2}{M}\sum_{m=0}^{M-1}e(m)X(m)cos\left(\dfrac{(2m+1)k\pi}{2M}\right)
\vspace{-5pt}
\end{equation}
\vspace{-5pt}

\noindent where
\vspace{-5pt}

\begin{equation*}
e(m) = 
\begin{cases}
    1/\sqrt{2}, & \text{if }m=0,\\
    1, & \text{otherwise}.
\end{cases}
\end{equation*}
\vspace{-5pt}

The inverse discrete cosine transform (IDCT) is defined as [1]
\vspace{-5pt}

\begin{equation}
X(m)=\dfrac{1}{\sqrt{2}}G_x(0)+\sum_{k=1}^{M-1}G_x(k)cos\left(\dfrac{(2m+1)k\pi}{2M}\right).
\end{equation}

\end{frame}

\section{Applying a Watermark}

\begin{frame}{Watermarking}

For every pixel in a block, the watermarked image block can be computed as
\vspace{-20pt}

\begin{equation}
G'(i,j)=G(i,j)+{\omega}W(i,j)
\end{equation}

\noindent where $\omega$ can be on the scale of $10^{-1}$ to $10^{-3}$.

\section{Comparing Image Similarity}

\begin{frame}{Structural Similarity Index}

The structural similarity index (SSIM) is defined as [7]

\begin{equation}
SSIM(i,j)=\dfrac{(2\mu_x\mu_y+C_1)(2\sigma_{xy}+C_2)}{(\mu_x^2+\mu_y^2+C_1)(\sigma_x^2+\sigma_y^2+C_2)}.
\end{equation}

\section{Extracting a Watermark}

\begin{frame}{Extracting a Watermark}

A watermark can be extracted with
\vspace{-15pt}

\begin{equation}
W'(x,y)=\dfrac{1}{\omega}\sum_{i=0}^{n-1}\sum_{j=1}^{m-1}[\delta_x\cdot\delta_y(G'(i,j)-G(i,j)],
\end{equation}
\vspace{-15pt}

\noindent where
\vspace{-15pt}

\begin{equation*}
\delta_x=
\begin{cases}
    1, & \text{if }i\text{ mod }u=x,\\
    0, & \text{otherwise},
\end{cases}
\end{equation*}
\vspace{-15pt}

\noindent and
\vspace{-15pt}

\begin{equation*}
\delta_y=
\begin{cases}
    1, & \text{if }j\text{ mod }v=y,\\
    0, & \text{otherwise}.
\end{cases}
\end{equation*}
\vspace{-20pt}
    
\end{frame}
